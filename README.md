# Instalación
## Termux:
$ pkg install git -y 

$ git clone https://github.com/Texsas-tool/texsas-ddos

$ cd biyivi_ataque_DDos

$ bash install.sh

$ python biyivi_ddos.py
## Linux:
$ git clone https://github.com/biyivi/biyivi_ataque_DDos

$ cd biyivi_ataque_DDos

$ python ddos.py
